2. Switch-Case & Goto (Simple Calculator)


#include
int main() {
int choice;
float num1, num2, result;
menu:
printf("\nSimple Calculator\n");
printf("1. Addition\n2. Subtraction\n3. Multiplication\n4. Division\n5. Exit\n");
printf("Enter your choice (1-5): ");
scanf("%d", &choice);
if (choice == 5) {
printf("Exiting the program...\n");
return 0;
}
printf("Enter two numbers: ");
scanf("%f %f", &num1, &num2);
switch(choice) {
case 1:
result = num1 + num2;
printf("Result: %.2f\n", result);
break;
case 2:
result = num1 - num2;
printf("Result: %.2f\n", result);
break;
case 3:
result = num1 * num2;
printf("Result: %.2f\n", result);
break;
case 4:
if (num2 == 0) {
printf("Error! Division by zero.\n");
goto menu;
}
result = num1 / num2;
printf("Result: %.2f\n", result);
break;
default:
printf("Invalid choice! Try again.\n");
goto menu;
}
goto menu;
}

3. Square Pattern Printing


#include
int main() {
int n;
printf("Enter the size of the square: ");
scanf("%d", &n);
for(int i = 0; i < n; i++) {
for(int j = 0; j < n; j++) {
printf("* ");
}
printf("\n");
}
return 0;
}

4. Right-Angled Pyramid Pattern


#include
int main() {
int n;
printf("Enter the height of the pyramid: ");
scanf("%d", &n);
for(int i = 1; i <= n; i++) {
for(int j = 1; j <= i; j++) {
printf("* ");
}
printf("\n");
}
return 0;
}
